#version 150

//uniform mat4 normalMatrix;
uniform sampler2D tex;

in vec2 fragTexCoord;
in vec3 fragColor;
in vec3 fragNormal;

out vec4 finalColor;

void main() {
    vec4 textureColor = texture(tex, fragTexCoord);
	finalColor = textureColor * vec4(fragColor, 1.0) * vec4(fragNormal, 1.0);
	finalColor = finalColor / vec4(fragNormal, 1.0);
}
